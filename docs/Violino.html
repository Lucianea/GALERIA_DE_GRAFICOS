dataf <- data.frame(xv, gv)
#plot
require(lattice)
## Carregando pacotes exigidos: lattice
library("latticeExtra")
## Carregando pacotes exigidos: RColorBrewer
bwplot(xv, panel = panel.violin, box.ratio = 3, xlab="y(units)")
